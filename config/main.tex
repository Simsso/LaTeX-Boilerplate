% line spacing between items in the table of contents and list of listings
% for the list of listings this needs to be placed after the code configuration (i.e. usepackage commands)
% partly taken from https://tex.stackexchange.com/questions/56546/how-to-change-spaces-between-items-in-table-of-contents
\usepackage{etoolbox}
\makeatletter
	\pretocmd{\section}{\addtocontents{toc}{\protect\addvspace{-5\p@}}}{}{}
	\pretocmd{\subsection}{\addtocontents{toc}{\protect\addvspace{-5\p@}}}{}{}
	\pretocmd{\subsubsection}{\addtocontents{toc}{\protect\addvspace{-5\p@}}}{}{}
	% line spacing between items in the list of listings (lol)
	\pretocmd{\lstinputlisting}{\addtocontents{lol}{\protect\addvspace{6\p@}}}{}{}
\makeatother
